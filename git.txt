git init - создать репозиторий
git add . - внести все файлы в коммит
git commit -m 'name' - создать коммит
git clone 'link' - скачать репозиторий
git status - логи последних изменений 
git push - добавляет все новые коммиты в репозиторий


--- 


git init                         # создать новый репозиторий в папке
git clone <url>                  # скопировать удалённый репозиторий
git add <файл>                   # добавить файл в отслеживаемые
git commit -m "описание"         # сохранить изменения
git push                         # отправить изменения на сервер
git pull                         # загрузить изменения с сервера
git checkout <ветка>             # переключиться на ветку
git merge <ветка>                # влить изменения из ветки
